# Permission Management Fix Implementation Plan

> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** Fix manage permission behavior so each role can only read or write the resources defined in `docs/superpowers/specs/2026-05-11-permission-management-fix-design.md`.

**Architecture:** Keep authorization predicates centralized in `internal/auth/authorizer.go`, then make services call those predicates for behavior enforcement. Frontend changes only mirror backend permissions by hiding disallowed actions; backend tests are the source of truth.

**Tech Stack:** Go 1.25, Gin handlers, sqlc query generation, testify, Vue 3, Pinia, TanStack Query, Naive UI, Vitest, browser validation through local dev server.

---

```go
// CanManageApp 判断主体是否可对应用执行写操作（如渠道绑定、运行时操作）。
// 平台管理员只有跨组织观察权，不继承应用写权限。
func CanManageApp(p Principal, appOrgID, appOwnerUserID string) bool {
	switch p.Role {
	case domain.UserRoleOrgAdmin:
		return p.OrgID == appOrgID
	case domain.UserRoleOrgMember:
		return p.UserID == appOwnerUserID
	default:
		return false
	}
}

// CanReadOrgKnowledge 判断主体是否可读取组织知识库。
func CanReadOrgKnowledge(p Principal, orgID string) bool {
	return CanViewOrg(p, orgID)
}

// CanWriteOrgKnowledge 判断主体是否可写入组织知识库。
func CanWriteOrgKnowledge(p Principal, orgID string) bool {
	return p.Role == domain.UserRoleOrgAdmin && p.OrgID == orgID
}

// CanRetryOrgKnowledgeSync 判断主体是否可触发组织知识库同步重试。
func CanRetryOrgKnowledgeSync(p Principal, orgID string) bool {
	return CanWriteOrgKnowledge(p, orgID)
}

// CanReadAppKnowledge 判断主体是否可读取指定应用的知识库。
func CanReadAppKnowledge(p Principal, appOrgID, appOwnerUserID string) bool {
	return CanViewApp(p, appOrgID, appOwnerUserID)
}

// CanWriteAppKnowledge 判断主体是否可写入指定应用的知识库。
func CanWriteAppKnowledge(p Principal, appOrgID, appOwnerUserID string) bool {
	return CanManageApp(p, appOrgID, appOwnerUserID)
}

// CanTriggerRuntimeOperation 判断主体是否可对应用触发运行时操作（启停/重启等）。
func CanTriggerRuntimeOperation(p Principal, appOrgID, appOwnerUserID string) bool {
	return CanManageApp(p, appOrgID, appOwnerUserID)
}

// CanCreateAppForOrg 判断主体是否可通过 onboarding 创建应用。
func CanCreateAppForOrg(p Principal, orgID string) bool {
	return p.Role == domain.UserRoleOrgAdmin && p.OrgID == orgID
}

// CanViewOrgUsage 判断主体是否可查看组织聚合用量。
func CanViewOrgUsage(p Principal, orgID string) bool {
	return p.Role == domain.UserRolePlatformAdmin || (p.Role == domain.UserRoleOrgAdmin && p.OrgID == orgID)
}

// CanViewMemberUsage 判断主体是否可查看成员用量。
func CanViewMemberUsage(p Principal, orgID, memberID string) bool {
	switch p.Role {
	case domain.UserRolePlatformAdmin:
		return true
	case domain.UserRoleOrgAdmin:
		return p.OrgID == orgID
	case domain.UserRoleOrgMember:
		return p.OrgID == orgID && p.UserID == memberID
	default:
		return false
	}
}

// CanViewOrgAudit 判断主体是否可查看组织审计列表。
func CanViewOrgAudit(p Principal, orgID string) bool {
	return p.Role == domain.UserRolePlatformAdmin || (p.Role == domain.UserRoleOrgAdmin && p.OrgID == orgID)
}

// CanViewOwnAudit 判断主体是否可查看自己的审计视角。
func CanViewOwnAudit(p Principal) bool {
	return p.UserID != ""
}
```

func (s *ChannelService) loadViewableApp(ctx context.Context, principal auth.Principal, appID string) (sqlc.App, error) {
	id, err := parseUUID(appID)
	if err != nil {
		return sqlc.App{}, ErrNotFound
	}
	app, err := s.store.GetApp(ctx, id)
	if errors.Is(err, pgx.ErrNoRows) {
		return sqlc.App{}, ErrNotFound
	}
	if err != nil {
		return sqlc.App{}, fmt.Errorf("查询应用失败: %w", err)
	}
	if !auth.CanViewApp(principal, uuidToString(app.OrgID), uuidToString(app.OwnerUserID)) {
		return sqlc.App{}, ErrForbidden
	}
	return app, nil
}

func (s *AuditService) ListMine(ctx context.Context, principal auth.Principal, limit, offset int32) ([]AuditResult, error) {
	if !auth.CanViewOwnAudit(principal) {
		return nil, ErrForbidden
	}
	actorID, err := parseUUID(principal.UserID)
	if err != nil {
		return nil, ErrForbidden
	}
	targetID := ""
	app, err := s.store.GetActiveAppByOwner(ctx, actorID)
	if err != nil && !errors.Is(err, pgx.ErrNoRows) {
		return nil, fmt.Errorf("查询个人应用失败: %w", err)
	}
	if err == nil {
		targetID = uuidToString(app.ID)
	}
	limit, offset = s.normalizePagination(limit, offset)
	rows, err := s.store.ListAuditLogsByActorOrAppTarget(ctx, sqlc.ListAuditLogsByActorOrAppTargetParams{
		ActorID:  actorID,
		TargetID: targetID,
		Limit:    limit,
		Offset:   offset,
	})
	if err != nil {
		return nil, fmt.Errorf("查询个人审计日志失败: %w", err)
	}
	return toAuditResults(rows), nil
}
