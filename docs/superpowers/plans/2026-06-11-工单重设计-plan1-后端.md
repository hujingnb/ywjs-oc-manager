# 定制技能工单重设计 · Plan 1 后端 Implementation Plan

> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** 后端把工单状态机改为动作驱动+前置校验,用统一消息表(text/image/file + JSON body)取代 comments+attachments,并支持交付后编辑可见范围。

**Architecture:** 破坏性迁移 000011(删 comments/attachments、建 skill_ticket_messages,body 为 MySQL JSON)→ sqlc 重生成 → service 重构(消息流 + 状态机校验 + targets 覆盖写)→ handler/router/DTO 调整 → OpenAPI 同步。不兼容历史数据。

**Tech Stack:** Go 1.26 / Gin / sqlc / golang-migrate / MySQL 8 JSON / guregu/null/v5 / testify。

- **sqlc 查询**:comment query 在 `internal/store/queries/skill_tickets.sql`;attachment query 在 `internal/store/queries/custom_skills.sql`(同文件还有 targets:`CreateCustomSkillTarget`/`DeleteCustomSkillTargetsByName`/`ListCustomSkillTargetsByName`)。
- **sqlc JSON 映射**:`sqlc.yaml` 已配置「可空 json → []byte」override;**非空 JSON 列默认映射为 `json.RawMessage`**(先例 `models.go` 的 `AssistantVersionIds`)。本表 `body JSON NOT NULL` → sqlc 生成 `Body json.RawMessage`。
- **blob**:`LibraryBlobStore.PutLibrarySkill(source, sourceRef, version, ext, data) (relPath, err)` / `OpenLibrarySkill(relPath) (io.ReadCloser, err)` / `DeleteLibrarySkill(relPath)`。附件原前缀 `ticket-attachment`;消息文件用 `ticket-message`。
- **错误常量**:`internal/service/errors.go`(`ErrSkillTicketDenied/ErrSkillTicketInvalid/ErrSkillTicketNotFound/ErrCustomSkill*`)。
- **路由装配**:`internal/api/router.go:193-202`,在 `user` 组(`RequireUserAuth`)注册;`Dependencies` 含 `SkillTicketService/CustomSkillService/SkillTicketAttachmentService`。
- **命令**:`make build`(编译);`make openapi-gen` + `make web-types-gen`(契约同步);`make openapi-check`(校验干净);`go test ./internal/...`(单测)。sqlc 重生成命令见 `Makefile`(查 `sqlc` target,通常 `make sqlc-gen` 或 `sqlc generate`)。

`internal/service/skill_ticket_message_service.go`:

```go
package service

import (
	"bytes"
	"context"
	"database/sql"
	"encoding/json"
	"errors"
	"fmt"
	"strings"
	"time"

	"oc-manager/internal/auth"
	"oc-manager/internal/store/sqlc"
)

// 消息类型常量。image/file 都存文件,按 content_type 前缀区分;text 存正文。
const (
	MessageKindText  = "text"
	MessageKindImage = "image"
	MessageKindFile  = "file"
)

// textPayload / filePayload 是 body JSON 的两种形状。
type textPayload struct {
	Text string `json:"text"`
}
type filePayload struct {
	ObjectPath  string `json:"object_path"`
	FileName    string `json:"file_name"`
	FileSize    int64  `json:"file_size"`
	ContentType string `json:"content_type"`
}

// SkillTicketMessageStore 是消息流所需数据能力。
type SkillTicketMessageStore interface {
	GetSkillTicket(ctx context.Context, id string) (sqlc.SkillTicket, error)
	UpdateSkillTicketStatus(ctx context.Context, arg sqlc.UpdateSkillTicketStatusParams) error
	TouchSkillTicket(ctx context.Context, id string) error
	CreateSkillTicketMessage(ctx context.Context, arg sqlc.CreateSkillTicketMessageParams) error
	ListSkillTicketMessages(ctx context.Context, ticketID string) ([]sqlc.SkillTicketMessage, error)
	GetSkillTicketMessage(ctx context.Context, id string) (sqlc.SkillTicketMessage, error)
}

// SkillTicketMessageService 管理工单统一消息流。
type SkillTicketMessageService struct {
	store SkillTicketMessageStore
	blobs LibraryBlobStore
}

func NewSkillTicketMessageService(store SkillTicketMessageStore, blobs LibraryBlobStore) *SkillTicketMessageService {
	return &SkillTicketMessageService{store: store, blobs: blobs}
}

// SkillTicketMessageResult 是消息对外视图(判别联合:按 kind 取不同字段)。object_path 不外泄,文件经下载端点取。
type SkillTicketMessageResult struct {
	ID           string    `json:"id"`
	AuthorUserID string    `json:"author_user_id"`
	Kind         string    `json:"kind"`
	Text         string    `json:"text,omitempty"`
	FileName     string    `json:"file_name,omitempty"`
	FileSize     int64     `json:"file_size,omitempty"`
	ContentType  string    `json:"content_type,omitempty"`
	CreatedAt    time.Time `json:"created_at"`
}

// loadTicketForMessage 取工单并做可见性校验(提交者本人或平台管理员)。
func (s *SkillTicketMessageService) loadTicketForMessage(ctx context.Context, p auth.Principal, ticketID string) (sqlc.SkillTicket, error) {
	row, err := s.store.GetSkillTicket(ctx, ticketID)
	if err != nil {
		if errors.Is(err, sql.ErrNoRows) {
			return sqlc.SkillTicket{}, ErrSkillTicketNotFound
		}
		return sqlc.SkillTicket{}, fmt.Errorf("查询工单失败: %w", err)
	}
	if !auth.CanViewSkillTicket(p, row.RequesterUserID) {
		return sqlc.SkillTicket{}, ErrSkillTicketDenied
	}
	return row, nil
}

// reopenIfRequesterOnClosed:提交者本人在 delivered/rejected 工单发消息 → 重开 pending;否则仅 Touch。
func (s *SkillTicketMessageService) reopenIfRequesterOnClosed(ctx context.Context, p auth.Principal, t sqlc.SkillTicket) error {
	closed := t.Status == SkillTicketStatusDelivered || t.Status == SkillTicketStatusRejected
	if p.UserID == t.RequesterUserID && closed {
		return s.store.UpdateSkillTicketStatus(ctx, sqlc.UpdateSkillTicketStatusParams{Status: SkillTicketStatusPending, ID: t.ID})
	}
	_ = s.store.TouchSkillTicket(ctx, t.ID)
	return nil
}

// SendText 发一条文本消息。
func (s *SkillTicketMessageService) SendText(ctx context.Context, p auth.Principal, ticketID, text string) (SkillTicketMessageResult, error) {
	t, err := s.loadTicketForMessage(ctx, p, ticketID)
	if err != nil {
		return SkillTicketMessageResult{}, err
	}
	body := strings.TrimSpace(text)
	if body == "" {
		return SkillTicketMessageResult{}, fmt.Errorf("%w: 消息不能为空", ErrSkillTicketInvalid)
	}
	raw, _ := json.Marshal(textPayload{Text: body})
	id := newUUID()
	if err := s.store.CreateSkillTicketMessage(ctx, sqlc.CreateSkillTicketMessageParams{
		ID: id, TicketID: ticketID, AuthorUserID: p.UserID, Kind: MessageKindText, Body: raw,
	}); err != nil {
		return SkillTicketMessageResult{}, fmt.Errorf("发送消息失败: %w", err)
	}
	if err := s.reopenIfRequesterOnClosed(ctx, p, t); err != nil {
		return SkillTicketMessageResult{}, fmt.Errorf("重开工单失败: %w", err)
	}
	return SkillTicketMessageResult{ID: id, AuthorUserID: p.UserID, Kind: MessageKindText, Text: body}, nil
}

// SendFile 发一条图片/文件消息(content_type 前缀 image/ → image,否则 file)。
func (s *SkillTicketMessageService) SendFile(ctx context.Context, p auth.Principal, ticketID, fileName, contentType string, data []byte) (SkillTicketMessageResult, error) {
	t, err := s.loadTicketForMessage(ctx, p, ticketID)
	if err != nil {
		return SkillTicketMessageResult{}, err
	}
	if len(data) == 0 {
		return SkillTicketMessageResult{}, fmt.Errorf("%w: 文件为空", ErrSkillTicketInvalid)
	}
	kind := MessageKindFile
	if strings.HasPrefix(contentType, "image/") {
		kind = MessageKindImage
	}
	id := newUUID()
	relPath, err := s.blobs.PutLibrarySkill("ticket-message", ticketID, id, safeExt(fileName), data)
	if err != nil {
		return SkillTicketMessageResult{}, fmt.Errorf("写入消息文件失败: %w", err)
	}
	raw, _ := json.Marshal(filePayload{ObjectPath: relPath, FileName: fileName, FileSize: int64(len(data)), ContentType: contentType})
	if err := s.store.CreateSkillTicketMessage(ctx, sqlc.CreateSkillTicketMessageParams{
		ID: id, TicketID: ticketID, AuthorUserID: p.UserID, Kind: kind, Body: raw,
	}); err != nil {
		_ = s.blobs.DeleteLibrarySkill(relPath)
		return SkillTicketMessageResult{}, fmt.Errorf("发送文件消息失败: %w", err)
	}
	if err := s.reopenIfRequesterOnClosed(ctx, p, t); err != nil {
		return SkillTicketMessageResult{}, fmt.Errorf("重开工单失败: %w", err)
	}
	return SkillTicketMessageResult{ID: id, AuthorUserID: p.UserID, Kind: kind, FileName: fileName, FileSize: int64(len(data)), ContentType: contentType}, nil
}

// ListMessages 列出工单消息并按 kind 解析 body。
func (s *SkillTicketMessageService) ListMessages(ctx context.Context, p auth.Principal, ticketID string) ([]SkillTicketMessageResult, error) {
	if _, err := s.loadTicketForMessage(ctx, p, ticketID); err != nil {
		return nil, err
	}
	rows, err := s.store.ListSkillTicketMessages(ctx, ticketID)
	if err != nil {
		return nil, fmt.Errorf("查询消息失败: %w", err)
	}
	out := make([]SkillTicketMessageResult, 0, len(rows))
	for _, r := range rows {
		out = append(out, toMessageResult(r))
	}
	return out, nil
}

// DownloadFile 取回图片/文件消息的原始字节(供下载端点);text 消息无文件。
func (s *SkillTicketMessageService) DownloadFile(ctx context.Context, p auth.Principal, ticketID, messageID string) ([]byte, string, string, error) {
	if _, err := s.loadTicketForMessage(ctx, p, ticketID); err != nil {
		return nil, "", "", err
	}
	m, err := s.store.GetSkillTicketMessage(ctx, messageID)
	if err != nil {
		if errors.Is(err, sql.ErrNoRows) {
			return nil, "", "", ErrSkillTicketNotFound
		}
		return nil, "", "", fmt.Errorf("查询消息失败: %w", err)
	}
	if m.TicketID != ticketID || (m.Kind != MessageKindImage && m.Kind != MessageKindFile) {
		return nil, "", "", fmt.Errorf("%w: 非文件消息", ErrSkillTicketInvalid)
	}
	var fp filePayload
	if err := json.Unmarshal(m.Body, &fp); err != nil {
		return nil, "", "", fmt.Errorf("解析文件消息失败: %w", err)
	}
	rc, err := s.blobs.OpenLibrarySkill(fp.ObjectPath)
	if err != nil {
		return nil, "", "", fmt.Errorf("打开消息文件失败: %w", err)
	}
	defer func() { _ = rc.Close() }()
	var buf bytes.Buffer
	if _, err := buf.ReadFrom(rc); err != nil {
		return nil, "", "", fmt.Errorf("读取消息文件失败: %w", err)
	}
	return buf.Bytes(), fp.FileName, fp.ContentType, nil
}

// toMessageResult 把一行消息按 kind 解析为对外视图。
func toMessageResult(r sqlc.SkillTicketMessage) SkillTicketMessageResult {
	out := SkillTicketMessageResult{ID: r.ID, AuthorUserID: r.AuthorUserID, Kind: r.Kind, CreatedAt: r.CreatedAt}
	switch r.Kind {
	case MessageKindText:
		var tp textPayload
		_ = json.Unmarshal(r.Body, &tp)
		out.Text = tp.Text
	case MessageKindImage, MessageKindFile:
		var fp filePayload
		_ = json.Unmarshal(r.Body, &fp)
		out.FileName, out.FileSize, out.ContentType = fp.FileName, fp.FileSize, fp.ContentType
	}
	return out
}
```

```go
func (s *SkillTicketService) transition(ctx context.Context, p auth.Principal, id, next string, allowed ...string) error {
	if !auth.CanManageSkillTicket(p) {
		return ErrSkillTicketDenied
	}
	row, err := s.loadTicket(ctx, id)
	if err != nil {
		return err
	}
	ok := false
	for _, a := range allowed {
		if row.Status == a {
			ok = true
			break
		}
	}
	if !ok {
		return fmt.Errorf("%w: 当前状态 %q 不允许此操作", ErrSkillTicketInvalid, row.Status)
	}
	if err := s.store.UpdateSkillTicketStatus(ctx, sqlc.UpdateSkillTicketStatusParams{Status: next, ID: id}); err != nil {
		return fmt.Errorf("更新工单状态失败: %w", err)
	}
	return nil
}

// StartProcessing 接单:pending → processing。
func (s *SkillTicketService) StartProcessing(ctx context.Context, p auth.Principal, id string) error {
```

```go
// UpdateTargets 覆盖写定制技能可见范围(交付后编辑)。仅平台管理员;工单须已交付。
func (s *CustomSkillService) UpdateTargets(ctx context.Context, p auth.Principal, ticketID string, targets []CustomSkillTargetInput) error {
	if !auth.CanManageSkillTicket(p) {
		return ErrCustomSkillDenied
	}
	if len(targets) == 0 {
		return fmt.Errorf("%w: 至少一个目标范围", ErrCustomSkillInvalid)
	}
	for _, tg := range targets {
		switch tg.Audience {
		case "all_org", "org_admins", "requester_only":
		default:
			return fmt.Errorf("%w: 非法受众 %q", ErrCustomSkillInvalid, tg.Audience)
		}
	}
	ticket, err := s.store.GetSkillTicket(ctx, ticketID)
	if err != nil {
		if errors.Is(err, sql.ErrNoRows) {
			return ErrSkillTicketNotFound
		}
		return fmt.Errorf("查询工单失败: %w", err)
	}
	if !ticket.CustomSkillName.Valid {
		return fmt.Errorf("%w: 工单尚未交付,无可见范围可编辑", ErrCustomSkillInvalid)
	}
	name := ticket.CustomSkillName.String
	// 覆盖写:先删后插(沿用 Deliver 的顺序写风格,无显式 tx)。
	if err := s.store.DeleteCustomSkillTargetsByName(ctx, name); err != nil {
		return fmt.Errorf("清空旧目标范围失败: %w", err)
	}
	for _, tg := range targets {
		if err := s.store.CreateCustomSkillTarget(ctx, sqlc.CreateCustomSkillTargetParams{
			ID: newUUID(), CustomSkillName: name, OrgID: tg.OrgID, Audience: tg.Audience,
		}); err != nil {
			return fmt.Errorf("写入目标范围失败: %w", err)
		}
	}
	return nil
}
```
