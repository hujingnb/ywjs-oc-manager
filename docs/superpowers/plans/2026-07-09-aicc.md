# AICC Implementation Plan

> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** Build the AICC (AI Contact Center) subsystem so org admins can create AI customer-service agents backed by hidden app/hermes runtimes, publish web chat entry points, collect leads, and review sessions/analytics.

**Architecture:** Manager owns AICC configuration, public visitor APIs, sessions, leads, feedback, rate limits, retention, and UI. Each AICC agent owns one hidden app/hermes runtime for actual AI responses, reusing the existing app provisioning, RAGFlow knowledge, new-api key, balance, and usage path. Anonymous visitor traffic always enters through manager public APIs; hermes is never exposed directly.

**Tech Stack:** Go 1.22+, Gin, sqlc, MySQL migrations, Redis, existing app/hermes oc-ops clients, S3/MinIO object storage, Vue 3 + TypeScript + Vite + TanStack Query + Naive UI, generated OpenAPI types.

```go
// AICC 能力 ----------------------------------------------------------

// CanManageAICCConfig 判断主体能否开通/关闭企业 AICC 能力并设置智能体数量上限。
// AICC 开通属于平台租户配置，只允许平台管理员维护。
func CanManageAICCConfig(p Principal) bool {
	return p.Role == domain.UserRolePlatformAdmin
}

// CanManageAICCAgent 判断主体能否管理指定企业的 AICC 智能体、投放、留资字段和运营数据。
// AICC 是企业侧客服业务，本期仅本企业 org_admin 可管理；平台管理员只保留只读排障能力。
func CanManageAICCAgent(p Principal, orgID string) bool {
	return p.Role == domain.UserRoleOrgAdmin && p.OrgID == orgID
}

// CanViewAICC 判断主体能否查看指定企业的 AICC 数据。
// 平台管理员可跨企业只读排障；企业管理员可看本企业；普通成员无 AICC 入口。
func CanViewAICC(p Principal, orgID string) bool {
	if p.Role == domain.UserRolePlatformAdmin {
		return true
	}
	return p.Role == domain.UserRoleOrgAdmin && p.OrgID == orgID
}
```

```go
// AICCService 负责 AICC 智能体管理与隐藏 app 绑定。
type AICCService struct {
	store AICCStore
	apps  AICCHiddenAppCreator
}

func NewAICCService(store AICCStore, apps AICCHiddenAppCreator) *AICCService {
	return &AICCService{store: store, apps: apps}
}

// CreateAgent 创建 AICC 智能体并自动创建隐藏 app。
func (s *AICCService) CreateAgent(ctx context.Context, p auth.Principal, input AICCAgentInput) (AICCAgentResult, error) {
	if p.OrgID == "" || !auth.CanManageAICCAgent(p, p.OrgID) {
		return AICCAgentResult{}, ErrForbidden
	}
	org, err := s.store.GetOrganization(ctx, p.OrgID)
	if err != nil {
		return AICCAgentResult{}, err
	}
	if !org.AiccEnabled {
		return AICCAgentResult{}, ErrForbidden
	}
	count, err := s.store.CountAICCAgentsByOrg(ctx, p.OrgID)
	if err != nil {
		return AICCAgentResult{}, err
	}
	if org.AiccAgentLimit.Valid && count >= org.AiccAgentLimit.Int64 {
		return AICCAgentResult{}, ErrQuotaExceeded
	}
	name := strings.TrimSpace(input.Name)
	if name == "" {
		return AICCAgentResult{}, fmt.Errorf("%w: AICC 智能体名称不能为空", ErrInvalidArgument)
	}
	retentionDays := input.RetentionDays
	if retentionDays == 0 {
		retentionDays = aiccDefaultRetentionDays
	}
	if retentionDays < 1 || retentionDays > aiccMaxRetentionDays {
		return AICCAgentResult{}, fmt.Errorf("%w: AICC 保留天数超出范围", ErrInvalidArgument)
	}
	agentID := newUUID()
	appID := newUUID()
	createdAppID, err := s.apps.CreateHiddenAICCApp(ctx, p, AICCHiddenAppInput{AppID: appID, OrgID: p.OrgID, UserID: p.UserID, Name: name})
	if err != nil {
		return AICCAgentResult{}, fmt.Errorf("创建 AICC 隐藏 app 失败: %w", err)
	}
	if createdAppID != "" {
		appID = createdAppID
	}
	if err := s.store.CreateAICCAgent(ctx, sqlc.CreateAICCAgentParams{
		ID:                 agentID,
		OrgID:              p.OrgID,
		AppID:              appID,
		Name:               name,
		Status:             domain.AICCAgentStatusDraft,
		Scenario:           nullStr(input.Scenario),
		Greeting:           nullStr(input.Greeting),
		AnswerBoundary:     nullStr(input.AnswerBoundary),
		PrivacyMode:        normalizeAICCPrivacyMode(input.PrivacyMode),
		PrivacyText:        nullStr(input.PrivacyText),
		RetentionDays:      retentionDays,
		ThemeJson:          input.ThemeJSON,
		AllowedDomainsJson: input.AllowedDomainsJSON,
		PublicToken:        newAICCToken(),
		WidgetToken:        newAICCToken(),
	}); err != nil {
		return AICCAgentResult{}, fmt.Errorf("创建 AICC 智能体失败: %w", err)
	}
	row, err := s.store.GetAICCAgent(ctx, agentID)
	if err != nil {
		return AICCAgentResult{}, err
	}
	return toAICCAgentResult(row), nil
}

func normalizeAICCPrivacyMode(mode string) string {
	if mode == domain.AICCPrivacyModeConsentRequired {
		return mode
	}
	return domain.AICCPrivacyModeNotice
}

func newAICCToken() string {
	var b [32]byte
	if _, err := rand.Read(b[:]); err != nil {
		panic(err)
	}
	return base64.RawURLEncoding.EncodeToString(b[:])
}
```

```go
// AICCHermesChat 抽象转发隐藏 app/hermes 的聊天能力。
type AICCHermesChat interface {
	ChatAICC(ctx context.Context, appID, sessionID, text string) (string, error)
}

type AICCPublicMessageInput struct {
	SessionToken string
	Text         string
	ImageFileID  string
}

type AICCPublicMessageResult struct {
	MessageID string `json:"message_id"`
	Text      string `json:"text"`
}

type AICCPublicService struct {
	store AICCPublicStore
	chat  AICCHermesChat
	now   func() time.Time
}

func NewAICCPublicService(store AICCPublicStore, chat AICCHermesChat) *AICCPublicService {
	return &AICCPublicService{store: store, chat: chat, now: time.Now}
}

// SendMessage 校验公开会话状态、隐私同意与必填留资后转发隐藏 app/hermes。
func (s *AICCPublicService) SendMessage(ctx context.Context, input AICCPublicMessageInput) (AICCPublicMessageResult, error) {
	session, err := s.store.GetAICCSessionByToken(ctx, strings.TrimSpace(input.SessionToken))
	if err != nil {
		return AICCPublicMessageResult{}, ErrAICCInvalidSession
	}
	agent, err := s.store.GetAICCAgent(ctx, session.AgentID)
	if err != nil {
		return AICCPublicMessageResult{}, ErrAICCOffline
	}
	if agent.Status != domain.AICCAgentStatusActive {
		return AICCPublicMessageResult{}, ErrAICCOffline
	}
	if agent.PrivacyMode == domain.AICCPrivacyModeConsentRequired && !session.PrivacyConsentedAt.Valid {
		return AICCPublicMessageResult{}, ErrAICCConsentRequired
	}
	missing, err := s.store.ListRequiredAICCLeadFieldsMissing(ctx, session.ID)
	if err != nil {
		return AICCPublicMessageResult{}, err
	}
	if len(missing) > 0 {
		return AICCPublicMessageResult{}, ErrAICCLeadRequired
	}
	text := strings.TrimSpace(input.Text)
	if text == "" && input.ImageFileID == "" {
		return AICCPublicMessageResult{}, fmt.Errorf("%w: 消息内容不能为空", ErrInvalidArgument)
	}
	visitorMessageID := newUUID()
	if err := s.store.CreateAICCMessage(ctx, sqlc.CreateAICCMessageParams{
		ID:          visitorMessageID,
		SessionID:   session.ID,
		AgentID:     session.AgentID,
		Direction:   "visitor",
		ContentType: "text",
		TextContent: nullStr(text),
	}); err != nil {
		return AICCPublicMessageResult{}, err
	}
	reply, err := s.chat.ChatAICC(ctx, agent.AppID, session.ID, text)
	if err != nil {
		return AICCPublicMessageResult{}, err
	}
	replyID := newUUID()
	if err := s.store.CreateAICCMessage(ctx, sqlc.CreateAICCMessageParams{
		ID:          replyID,
		SessionID:   session.ID,
		AgentID:     session.AgentID,
		Direction:   "assistant",
		ContentType: "text",
		TextContent: nullStr(reply),
	}); err != nil {
		return AICCPublicMessageResult{}, err
	}
	return AICCPublicMessageResult{MessageID: replyID, Text: reply}, nil
}
```

- [ ] **Step 3: Implement retention service**

Create `internal/service/aicc_retention.go`:

```go
package service

import "context"

type AICCRetentionStore interface {
	ListExpiredAICCSessions(ctx context.Context, limit int32) ([]sqlc.AiccSession, error)
	ListAICCImageObjectKeysBySession(ctx context.Context, sessionID string) ([]string, error)
	DeleteAICCSession(ctx context.Context, id string) error
}

type AICCObjectCleaner interface {
	DeleteObject(ctx context.Context, key string) error
}

type AICCRetentionService struct {
	store AICCRetentionStore
	blob  AICCObjectCleaner
}

func NewAICCRetentionService(store AICCRetentionStore, blob AICCObjectCleaner) *AICCRetentionService {
	return &AICCRetentionService{store: store, blob: blob}
}

// CleanupExpiredSessions 删除超过保留期的 AICC 会话及其图片对象。
func (s *AICCRetentionService) CleanupExpiredSessions(ctx context.Context, limit int32) (int64, error) {
	if limit <= 0 || limit > 500 {
		limit = 100
	}
	sessions, err := s.store.ListExpiredAICCSessions(ctx, limit)
	if err != nil {
		return 0, err
	}
	var deleted int64
	for _, session := range sessions {
		keys, err := s.store.ListAICCImageObjectKeysBySession(ctx, session.ID)
		if err != nil {
			return deleted, err
		}
		for _, key := range keys {
			if err := s.blob.DeleteObject(ctx, key); err != nil {
				return deleted, err
			}
		}
		if err := s.store.DeleteAICCSession(ctx, session.ID); err != nil {
			return deleted, err
		}
		deleted++
	}
	return deleted, nil
}
```
